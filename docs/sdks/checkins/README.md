## list

### Example Usage

```typescript
import { Pushpress } from "pushpress";

const pushpress = new Pushpress({
  apiKey: process.env["PUSHPRESS_API_KEY"] ?? "",
});

async function run() {
  const result = await pushpress.checkins.list({});

  for await (const page of result) {
    // Handle the page
    console.log(page);
  }
}

run();
```
